# 4 Настройте межсетевой экран на маршрутизаторах HQ-RTR и BR-RTR на сеть в сторону ISP
### 4.1 Обеспечьте работу протоколов http, https, dns, ntp, icmp или дополнительных нужных протоколов
### 4.2 Запретите остальные подключения из сети Интернет во внутреннюю сеть.

 *Настройка HQ-RTR*
 ```bash
en
conf t
filter-map ipv4 ACL 10
match icmp 172.16.1.1/28 192.168.0.1/26
match 80 172.16.1.1/28 192.168.0.1/26
match 123 172.16.1.1/28 192.168.0.1/26
match 53 172.16.1.1/28 192.168.0.1/26
match tcp 172.16.1.1/28 192.168.0.1/26 eq 443
set accept
ex
filter-map ipv4 ACLR 20
match any 172.16.1.1/28 192.168.0.1/26
set reject
end
wr mem
```
