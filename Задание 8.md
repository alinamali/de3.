# 8 Реализуйте механизм инвентаризации машин HQ-SRV и HQ-CLI через Ansible на BR-SRV:
### 8.1 Плейбук должен собирать информацию о рабочих местах: Имя компьютера, IP-адрес компьютера
### 8.2 Плейбук, должен быть размещен в директории /etc/ansible, отчёты в поддиректории PC-INFO, в формате .yml. Файлы должны называться именем компьютера, который был инвентаризирован
### 8.3 Файл плейбука располагается в образе Additional.iso в директории playbook


*Настройка BR-SRV*

```bash
find / -name "Additional.iso" 2>/dev/null

mount -o loop /dev/disk/by-label/Additional.iso /mnt/iso

mcedit /etc/ansible/hosts

[HQ]
HQ-SRV ansible_host=192.168.0.10 ansible_user=sshuser ansible_password=P@ssw0rd ansible_port=2026
HQ-CLI ansible_host=192.168.1.3 ansible_user=sshuser ansible_password=P@ssw0rd ansible_port=2026
[RTR]
HQ-RTR ansible_host=10.10.10.1 или 172.16.0.1 ansible_user=admin ansible_password=admin ansible_connection=network_cli ansible_network_os=ios
BR-RTR ansible_host=192.168.3.1 ansible_user=admin ansible_password=admin ansible_connection=network_cli ansible_network_os=ios
[all:vars]
ansible_python_interpreter=/usr/bin/python3



mcedit /etc/ansible/PC_INFO/playbook.yml


- name: Инвентаризация машин HQ-SRV и HQ-CLI
  hosts: HQ
  gather_facts: yes
  become: no

  tasks:
    - name: Создание директории для отчетов
      file:
        path: /etc/ansible/PC-INFO
        state: directory
        mode: '0755'
      delegate_to: localhost
      run_once: yes

    - name: Сбор информации о хосте
      set_fact:
        pc_name: "{{ ansible_hostname }}"
        pc_ip: "{{ ansible_default_ipv4.address }}"

    - name: Сохранение информации в файл
      copy:
        content: |
          pc_name: "{{ pc_name }}"
          pc_ip: "{{ pc_ip }}"
          gathered_at: "{{ ansible_date_time.iso8601 }}"
          os: "{{ ansible_distribution }} {{ ansible_distribution_version }}"
        dest: "/etc/ansible/PC-INFO/{{ pc_name }}.yml"
        mode: '0644'
      delegate_to: localhost

    - name: Вывод результата
      debug:
        msg: "Информация о {{ ansible_hostname }} сохранена в /etc/ansible/PC-INFO/{{ ansible_hostname }}.yml"




ansible-playbook -i /etc/ansible/hosts /etc/ansible/PC_INFO/playbook.yml

ls /etc/ansible/PC_INFO/
ls -la /etc/ansible/PC_INFO/
```
